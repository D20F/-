## 解构[1,2,["3",4,"5",[6,[7,8],9]]]不改变数据类型
``` js
{
  function parseArr(arr,res){
    for(var i=0;i<arr.length;i++){
      if(arr[i] instanceof Array){
        parseArr(arr[i],res);
      }else{
        res.push(arr[i]);
      }
    }
  }
  var a=[1,2,["3",4,"5",[6,[7,8],9]]]
  var res=[];
  parseArr(a,res);
}
```
